\begin{Problema}{1}
  Calcular el valor de
  \begin{equation}
    \label{eq:1}
    \sqrt{1+3+5+7+\cdots+2003+2005+2007},
  \end{equation}
  donde la suma dentro de la raíz cuadrada es la suma de todos los
  números impares del 1 al 2007.
\end{Problema}

\begin{Solucion}
  La suma $1+3+\cdots+(2n-1)$ de los primeros $n$ números impares es
  igual a $n^{2}$. Si $2n-1=2007$ entonces $n=1004$, por lo que la
  suma~\ref{eq:1} vale $1004^{2}$.
\end{Solucion}
